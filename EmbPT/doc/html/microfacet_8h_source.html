00130     __forceinline Col3f eval(const Vec3f& wo, const DifferentialGeometry& dg, const Vec3f& wi) const
00131     {
00132       float cosThetaO = dot(wo,dg.Ns);
00133       float cosThetaI = dot(wi,dg.Ns);
00134       if (cosThetaI <= 0.0f || cosThetaO <= 0.0f) return zero;
00135       Vec3f wh = normalize(wi + wo);
00136       float cosThetaH = dot(wh, dg.Ns);
00137       float cosTheta = dot(wi, wh); // = dot(wo, wh);
00138       Col3f F = fresnel.eval(cosTheta);
00139       float D = distribution.eval(cosThetaH);
00140       float G = min(1.0f, 2.0f * cosThetaH * cosThetaO / cosTheta, 2.0f * cosThetaH * cosThetaI / cosTheta);
00141       return R * D * G * F / (4.0f*cosThetaO);
00142     }
00143 
00144     Col3f sample(const Vec3f& wo, const DifferentialGeometry& dg, Sample3f& wi, const Vec2f& s) const
00145     {
00146       if (dot(wo,dg.Ns) <= 0.0f) return zero;
00147       distribution.sample(wo,dg,wi,s);
00148       if (dot(wi.value,dg.Ns) <= 0.0f) return zero;
00149       return eval(wo,dg,wi);
00150     }
00151 
00152     float pdf(const Vec3f& wo, const DifferentialGeometry& dg, const Vec3f& wi) const {

00154     }
