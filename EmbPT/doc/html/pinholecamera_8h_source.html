00030     PinholeCamera(const Parms& parms) {
00031       localToWorld      = parms.getTransform("local2world");
00032       float angle       = parms.getFloat("angle",64.0f);
00033       float aspectRatio = parms.getFloat("aspectRatio",1.0f);
00034       lensRadius        = parms.getFloat("lensRadius",0.0f);
00035       focalDistance     = parms.getFloat("focalDistance");
00036       Vec3f W           = xfmVector(localToWorld, Vec3f(-0.5f*aspectRatio,-0.5f,0.5f/tanf(deg2rad(0.5f*angle))));
00037       xfm = AffineSpace(aspectRatio*localToWorld.l.vx,localToWorld.l.vy,W,localToWorld.p);
00038     }
00039 
00040     void ray(const Vec2f& pixel, const Vec2f& sample, Ray& ray_o) const 
00041     {
00042       // Special case for pinhole camera and rays through center of lens.
00043       if (lensRadius == 0.0f || sample == Vec2f(0.5f,0.5f)) {
00044         new (&ray_o) Ray(xfm.p,normalize(pixel.x*xfm.l.vx + (1.0f-pixel.y)*xfm.l.vy + xfm.l.vz));
00045         return;
00046       }
00047 
00048       // Sample uniform location on a disc with r = lensRadius
00049       float r, theta;
00050       Vec2f s = 2.0f * sample - Vec2f(1.0f, 1.0f);
00051 
00052       if (s.x >= -s.y) {
00053         if (s.x > s.y) {
00054           r = s.x;
00055           if (s.y > 0.0) theta = s.y/r;
00056           else theta = 8.0f + s.y/r;
00057         }
00058         else {
00059           r = s.y;
00060           theta = 2.0f - s.x/r;
00061         }
00062       }
00063       else {
00064         if (s.x <= s.y) {
00065           r = -s.x;
00066           theta = 4.0f - s.y/r;
00067         }
00068         else {
00069           r = -s.y;
00070           theta = 6.0f + s.x/r;
00071         }
00072       }
00073       theta *= float(pi) / 4.0f;
00074 
00075       Vec3f org = xfmPoint(localToWorld, Vec3f(lensRadius*r*cosf(theta), lensRadius*r*sinf(theta), 0.0f));
00076       Vec3f dir = normalize(pixel.x*xfm.l.vx + (1.0f-pixel.y)*xfm.l.vy + xfm.l.vz);
00077       Vec3f centerDir = normalize(0.5f*xfm.l.vx + 0.5f*xfm.l.vy + xfm.l.vz);
00078       Vec3f p = xfm.p + focalDistance*dir/dot(dir,centerDir);
00079       new (&ray_o) Ray(org, normalize(p - org));
00080       return;
00081     }
00082 
00083   private:
00084 
00087     AffineSpace xfm;
00088     AffineSpace localToWorld;
00089     float lensRadius;
00090     float focalDistance;
00091   };
